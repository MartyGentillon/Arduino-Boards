64 void ConsumerControl_::press(uint16_t m) {
65  // search for a free spot
66  for (uint8_t i = 0; i < sizeof(HID_ConsumerControlReport_Data_t) / 2; i++) {
67  if (_report.keys[i] == 0x00) {

74 void ConsumerControl_::release(uint16_t m) {
75  // search and release the keypress

77  if (_report.keys[i] == m) {

80  }
81  }

92 void ConsumerControl_::sendReport(void) {
93  // If the last report is different than the current report, then we need to send a report.
94  // We guard sendReport like this so that calling code doesn't end up spamming the host with empty reports
95  // if sendReport is called in a tight loop.
96 
97  // if the previous report is the same, return early without a new report.
98  if (memcmp(&_lastReport, &_report, sizeof(_report)) == 0)
99  return;
100 
101  sendReportUnchecked();
102  memcpy(&_lastReport, &_report, sizeof(_report));
103 }
