43 void handleKeyswitchEvent(Key mappedKey, byte row, byte col, uint8_t keyState) {
44  /* These first steps are only done for keypresses that have a real (row,col).
45  * In particular, doing them for keypresses with out-of-bounds (row,col)
46  * would cause out-of-bounds array accesses in Layer.lookup(),
47  * Layer.updateLiveCompositeKeymap(), etc.
48  * Perhaps this check should instead be for INJECTED - I'm not clear on
49  * whether keypresses with out-of-bounds (row,col) are the same as
50  * keypresses with INJECTED set, or are a superset or a subset of
51  * INJECTED keypresses. In any case, the (row,col) test is *safe* in that
52  * it avoids out-of-bounds accesses, at least in core. (Can't promise
53  * anything about event handlers - they may still receive out-of-bounds
54  * (row,col), and handling that properly is on them.)
55  */
56  if (row < ROWS && col < COLS) {
57 
58  /* If a key had an on event, we update the live composite keymap. See
59  * layers.h for an explanation about the different caches we have. */
60  if (keyToggledOn(keyState))
61  Layer.updateLiveCompositeKeymap(row, col);
62 
63  /* If the key we are dealing with is masked, ignore it until it is released.
64  * When releasing it, clear the mask, so future key events can be handled
65  * appropriately.
66  *
67  * See layers.cpp for an example that masks keys, and the reason why it does
68  * so.
69  */
70  if (KeyboardHardware.isKeyMasked(row, col)) {
71  if (keyToggledOff(keyState)) {
72  KeyboardHardware.unMaskKey(row, col);
73  } else {
74  return;
75  }
76  }
77 
78  if (!(keyState & INJECTED)) {
79  mappedKey = Layer.lookup(row, col);
80  }
81 
82  } // row < ROWS && col < COLS
83 
84  // Keypresses with out-of-bounds (row,col) start here in the processing chain
85 
86  for (byte i = 0; Kaleidoscope.eventHandlers[i] != NULL && i < HOOK_MAX; i++) {
87  Kaleidoscope_::eventHandlerHook handler = Kaleidoscope.eventHandlers[i];
88  mappedKey = (*handler)(mappedKey, row, col, keyState);
89  if (mappedKey.raw == Key_NoKey.raw)
90  return;
91  }
92  mappedKey = Layer.eventHandler(mappedKey, row, col, keyState);
93  if (mappedKey.raw == Key_NoKey.raw)
94  return;
95  handleKeyswitchEventDefault(mappedKey, row, col, keyState);
96 }
