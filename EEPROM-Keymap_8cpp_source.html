39 Key EEPROMKeymap::getKey(uint8_t layer, byte row, byte col) {
40  Key key;
41 
42  if (layer >= max_layers_)
43  return Key_NoKey;
44 
45  uint16_t pos = ((layer * ROWS * COLS) + (row * COLS) + col) * 2;
46 
47  key.flags = EEPROM.read(keymap_base_ + pos);
48  key.keyCode = EEPROM.read(keymap_base_ + pos + 1);
49 
50  return key;
51 }

83 bool EEPROMKeymap::focusKeymap(const char *command) {
84  if (strcmp_P(command, PSTR("keymap.map")) != 0)
85  return false;
86 
87  if (Serial.peek() == '\n') {
88  for (uint8_t layer = 0; layer < max_layers_; layer++) {
89  for (uint8_t row = 0; row < ROWS; row++) {
90  for (uint8_t col = 0; col < COLS; col++) {
91  Key k = Layer.getKey(layer, row, col);
92 
93  printKey(k);
94  ::Focus.printSpace();
95  }
96  }
97  }
98  Serial.println();
99  } else {
100  uint16_t i = 0;
101  while ((Serial.peek() != '\n') && (i < ROWS * COLS * max_layers_)) {
102  updateKey(i, parseKey());
103  i++;
104  }
105  }
106 
107  return true;
108 }

110 bool EEPROMKeymap::focusKeymapLayer(const char *command) {
111  if (strcmp_P(command, PSTR("keymap.layer")) != 0) {
112  return false;
113  }
114 
115  uint8_t layer = Serial.parseInt();
116  if (layer >= max_layers_) {
117  return false;
118  }
119  if (Serial.peek() == '\n') {
120  for (uint8_t row = 0; row < ROWS; row++) {
121  for (uint8_t col = 0; col < COLS; col++) {
122  Key k = Layer.getKey(layer, row, col);
123  printKey(k);
124  ::Focus.printSpace();
125  }
126  }
127  } else {
128  uint16_t keysPerLayer = ROWS * COLS;
129  uint16_t offset = layer * keysPerLayer;
130  for (uint16_t k = 0; (k < keysPerLayer) && (Serial.peek() != '\n'); k++) {
131  updateKey(offset + k, parseKey());
132  }
133  }
134 
135  return true;
136
