28 int HID_::getInterface(uint8_t* interfaceCount) {
29  *interfaceCount += 1; // uses 1
30  HIDDescriptor hidInterface = {
31  D_INTERFACE(pluggedInterface, 1, USB_DEVICE_CLASS_HUMAN_INTERFACE, HID_SUBCLASS_NONE, HID_PROTOCOL_NONE),
32  D_HIDREPORT(descriptorSize),

35  return USB_SendControl(0, &hidInterface, sizeof(hidInterface));

38 int HID_::getDescriptor(USBSetup& setup) {
39  // Check if this is a HID Class Descriptor request
40  if (setup.bmRequestType != REQUEST_DEVICETOHOST_STANDARD_INTERFACE) {
41  return 0;
42  }
43  if (setup.wValueH != HID_REPORT_DESCRIPTOR_TYPE) {
44  return 0;
45  }
46 
47  // In a HID Class Descriptor wIndex cointains the interface number
48  if (setup.wIndex != pluggedInterface) {
49  return 0;
50  }
51 
52  int total = 0;
53  HIDSubDescriptor* node;
54  for (node = rootNode; node; node = node->next) {
55  int res = USB_SendControl(TRANSFER_PGM, node->data, node->length);
56  if (res == -1)
57  return -1;
58  total += res;
59  }
60 
61  // Reset the protocol on reenumeration. Normally the host should not assume the state of the protocol
62  // due to the USB specs, but Windows and Linux just assumes its in report mode.
63  protocol = HID_REPORT_PROTOCOL;
64 
65  return total;
66 }

91 int HID_::SendReport(uint8_t id, const void* data, int len) {
92  auto ret = USB_Send(pluggedEndpoint, &id, 1);
93  if (ret < 0) return ret;
94  auto ret2 = USB_Send(pluggedEndpoint | TRANSFER_RELEASE, data, len);
95  if (ret2 < 0) return ret2;
96  return ret + ret2;
97 }

99 bool HID_::setup(USBSetup& setup) {
100  if (pluggedInterface != setup.wIndex) {
101  return false;
102  }
103 
104  uint8_t request = setup.bRequest;
105  uint8_t requestType = setup.bmRequestType;
106 
107  if (requestType == REQUEST_DEVICETOHOST_CLASS_INTERFACE) {
108  if (request == HID_GET_REPORT) {
109  // TODO: HID_GetReport();
110  return true;
111  }
112  if (request == HID_GET_PROTOCOL) {
113  // TODO: Send8(protocol);
114  return true;
115  }
116  if (request == HID_GET_IDLE) {
117  // TODO: Send8(idle);
118  }
119  }
120 
121  if (requestType == REQUEST_HOSTTODEVICE_CLASS_INTERFACE) {
122  if (request == HID_SET_PROTOCOL) {
123  // The USB Host tells us if we are in boot or report mode.
124  // This only works with a real boot compatible device.
125  protocol = setup.wValueL;
126  return true;
127  }
128  if (request == HID_SET_IDLE) {
129  idle = setup.wValueL;
130  return true;
131  }
132  if (request == HID_SET_REPORT) {
133  uint16_t length = setup.wLength;
134 
135  if (length == sizeof(setReportData)) {
136  USB_RecvControl(&setReportData, length);
137  } else if (length == sizeof(setReportData.leds)) {
138  USB_RecvControl(&setReportData.leds, length);
139  setReportData.reportId = 0;
140  }
141  }
142  }
143 
144  return false;
145 }
